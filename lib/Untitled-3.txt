import 'package:dartz/dartz.dart';
import '../entities/file_material.dart';
import '../entities/failure.dart';
import '../repositories/file_repository.dart';

class GetFilesForStudent {
  final FileRepository repository;

  GetFilesForStudent(this.repository);

  Future<Either<Failure, List<FileMaterial>>> call({
    required String studentId,
    String? subject,
    List<String>? tags,
  }) {
    return repository.getFilesForStudent(
      studentId: studentId,
      subject: subject,
      tags: tags,
    );
  }
}
